### Query Builder

```lua
local orm = require("orm")

-- Basic SELECT
local users = orm.safeIdentifier("users")
local q = orm.from(users)
    :where(where_fragment)
    :orderBy(orm.safeIdentifier("created_at"), true)
    :limit(50)
    :toSql()
    :toString()
-- => SELECT * FROM users WHERE ... ORDER BY created_at ASC LIMIT 50

-- WHERE convenience methods
orm.from(users)
    :whereNull(orm.safeIdentifier("deleted_at"))
    :whereNotNull(orm.safeIdentifier("email"))
    :whereIn(orm.safeIdentifier("role"), {orm.SqlString("admin"), orm.SqlString("editor")})
    :whereNot(fragment)
    :whereBetween(orm.safeIdentifier("age"), orm.SqlInt32(18), orm.SqlInt32(65))
    :whereLike(orm.safeIdentifier("name"), "%smith%")
    :whereILike(orm.safeIdentifier("email"), "%@gmail.com")
    :toSql()

-- OR conditions
orm.from(users)
    :where(condition1)
    :orWhere(condition2)
    :toSql()

-- SELECT specific columns
orm.from(users)
    :select({orm.safeIdentifier("id"), orm.safeIdentifier("name")})
    :toSql()

-- SELECT with expressions
orm.from(users)
    :selectExpr({orm.countAll()})
    :toSql()

-- JOINs
local posts = orm.safeIdentifier("posts")
orm.from(users)
    :join("inner", posts, on_condition)       -- INNER JOIN
    :join("left", posts, on_condition)        -- LEFT JOIN
    :join("right", posts, on_condition)       -- RIGHT JOIN
    :join("full", posts, on_condition)        -- FULL OUTER JOIN
    :crossJoin(posts)                         -- CROSS JOIN
    :toSql()

-- Column references for JOINs
local user_id_col = orm.col(users, orm.safeIdentifier("id"))
local post_user_id_col = orm.col(posts, orm.safeIdentifier("user_id"))

-- GROUP BY / HAVING / DISTINCT
orm.from(users)
    :select({orm.safeIdentifier("role")})
    :selectExpr({orm.countAll()})
    :groupBy({orm.safeIdentifier("role")})
    :having(having_fragment)
    :orHaving(or_having_fragment)
    :distinct()
    :toSql()

-- OFFSET
orm.from(users)
    :limit(10)
    :offset(20)
    :toSql()

-- Row locking
orm.from(users)
    :where(condition)
    :lock(orm.ForUpdate())    -- FOR UPDATE
    :toSql()

orm.from(users)
    :where(condition)
    :lock(orm.ForShare())     -- FOR SHARE
    :toSql()

-- ORDER BY with NULLS FIRST/LAST
orm.from(users)
    :orderByNulls(orm.safeIdentifier("score"), true, orm.NullsFirst())
    :orderByNulls(orm.safeIdentifier("name"), false, orm.NullsLast())
    :toSql()

-- Safe query with default limit (CWE-400 protection)
orm.from(users)
    :where(condition)
    :safeToSql(100)
    :toString()

-- Count query
orm.from(users)
    :where(condition)
    :countSql()
    :toString()
```

### Changeset Pipeline

local orm = require("orm")

-- Define schema
local users_def = orm.TableDef(
    orm.safeIdentifier("users"),
    {
        orm.FieldDef(orm.safeIdentifier("name"), orm.StringField(), false, nil, false),
        orm.FieldDef(orm.safeIdentifier("email"), orm.StringField(), false, nil, false),
        orm.FieldDef(orm.safeIdentifier("age"), orm.IntField(), true, nil, false),
        orm.FieldDef(orm.safeIdentifier("active"), orm.BoolField(), false, nil, false),
        -- add timestamps() entries for inserted_at/updated_at
    },
    orm.safeIdentifier("id")   -- custom primary key
)

-- Build a changeset from user input
local params = {name = "Alice", email = "alice@example.com", age = "30"}
local cs = orm.changeset(users_def, params)
    :cast({orm.safeIdentifier("name"), orm.safeIdentifier("email"), orm.safeIdentifier("age")})
    :validateRequired({orm.safeIdentifier("name"), orm.safeIdentifier("email")})
    :validateLength(orm.safeIdentifier("name"), 1, 100)
    :validateInt(orm.safeIdentifier("age"))

-- Check validity
if cs.isValid then
    local insert_sql = cs:toInsertSql():toString()
    -- => INSERT INTO users (name, email, age) VALUES ('Alice', 'alice@example.com', 30)
end

-- Update existing record
local update_sql = cs:toUpdateSql(orm.SqlInt32(42)):toString()
-- => UPDATE users SET name = 'Alice', email = 'alice@example.com', age = 30 WHERE id = 42

-- Additional validations
cs:validateInt64(orm.safeIdentifier("big_num"))
  :validateFloat(orm.safeIdentifier("score"))
  :validateBool(orm.safeIdentifier("active"))
  :validateInclusion(orm.safeIdentifier("role"), {"admin", "user", "guest"})
  :validateExclusion(orm.safeIdentifier("status"), {"banned"})
  :validateNumber(orm.safeIdentifier("age"), 0, 150)
  :validateAcceptance(orm.safeIdentifier("terms"))
  :validateConfirmation(orm.safeIdentifier("password"), orm.safeIdentifier("password_confirmation"))
  :validateContains(orm.safeIdentifier("bio"), "keyword")
  :validateStartsWith(orm.safeIdentifier("code"), "PRE-")
  :validateEndsWith(orm.safeIdentifier("email"), ".com")

-- Data manipulation
cs:putChange(orm.safeIdentifier("updated_at"), "2026-03-13")
  :getChange(orm.safeIdentifier("name"))
  :deleteChange(orm.safeIdentifier("age"))

## Security Model

The Alloy ORM is built on a defense-in-depth approach to SQL injection prevention, using Temper's type system to enforce safety at compile time. The compiled Lua output preserves all safety invariants.

### Defense Layers

**Layer 1: SafeIdentifier (CWE-89 -- table/column names)**

Table and column names must pass through `safeIdentifier()`, which validates against `[a-zA-Z_][a-zA-Z0-9_]*`. The internal `ValidatedIdentifier` implementation is not exported -- external code cannot construct one without validation. This is the only path to `appendSafe` at runtime, closing off identifier-based injection entirely.

```lua
orm.safeIdentifier("users")        -- => ValidatedIdentifier (OK)
orm.safeIdentifier("users; DROP")  -- => error (rejected)
orm.safeIdentifier("table--name")  -- => error (rejected)
```

**Layer 2: Typed SqlPart hierarchy (CWE-89 -- value escaping)**

User values never touch `appendSafe`. Each type gets its own `SqlPart` implementation with type-appropriate rendering:

| Type | Rendering | Example |
|------|-----------|---------|
| `SqlString` | Quote-wraps with `'` escaping (`'` becomes `''`) | `'O''Brien'` |
| `SqlInt32` / `SqlInt64` | Bare integer via `tostring()` | `42` |
| `SqlBoolean` | Literal `TRUE` / `FALSE` | `TRUE` |
| `SqlFloat64` | Bare float (NaN/Infinity render as `NULL`) | `3.14` |
| `SqlDate` | Quote-wrapped date string with `'` escaping | `'2026-03-13'` |
| `SqlDefault` | Literal `DEFAULT` keyword | `DEFAULT` |

The `SqlPart` interface is sealed -- the compiler enforces exhaustive handling.

**Layer 3: SqlBuilder separation (CWE-89 -- structural safety)**

`SqlBuilder` enforces a clean separation between SQL structure and data:
- `appendSafe(str)` -- for SQL keywords and validated identifiers only
- `appendString(val)` -- wraps in `SqlString` (escaped)
- `appendInt32(val)` -- wraps in `SqlInt32` (bare integer)

There is no `appendRaw` or bypass method.

**Layer 4: Changeset pipeline (CWE-915 -- mass assignment)**

`Changeset` is a sealed interface -- `ChangesetImpl` is not exported, so construction only happens through the `changeset()` factory. The pipeline enforces:
- `cast(allowedFields)` -- field whitelisting, prevents mass assignment
- `validateRequired(fields)` -- non-nullable enforcement
- `toInsertSql()` / `toUpdateSql(id)` -- independently enforces non-nullable fields
- Raw `params` are never exposed -- only whitelisted `changes` are readable

**Layer 5: Query builder (CWE-89, CWE-400)**

`Query` requires `SafeIdentifier` for table names, column selections, ORDER BY clauses, and JOIN table names. WHERE and ON conditions accept only `SqlFragment` (never raw strings). `safeToSql(defaultLimit)` enforces result set bounds (CWE-400). `UpdateQuery` and `DeleteQuery` error on empty WHERE clauses, preventing accidental full-table mutations.

### Per-Phase Security Analysis

**JOIN Feature Security** -- JOIN keywords use a sealed `JoinType` interface with hardcoded strings. Table names require `SafeIdentifier`. ON conditions accept only `SqlFragment`. The `col()` helper produces qualified column references from two `SafeIdentifier` values joined by a hardcoded `.` separator.

**WHERE Clause Enrichment** -- `WhereClause` is sealed (`AndCondition`/`OrCondition`). Convenience methods (`whereNull`, `whereIn`, `whereBetween`, `whereLike`, `whereILike`) build fragments internally using only validated identifiers and typed values. Empty IN list produces `1 = 0`. LIKE/ILIKE patterns go through `SqlString` escaping.

**Aggregation** -- All 6 aggregate functions use hardcoded function names via `appendSafe`. Field names require `SafeIdentifier`. `selectExpr`, `groupBy`, `having`/`orHaving` reuse existing safe types. `distinct()` selects between two hardcoded strings.

**Set Operations and Subqueries** -- Set operation keywords (`UNION`, `INTERSECT`, `EXCEPT`) are hardcoded literals. Both operands are `Query` objects. Sub-SELECTs are parenthesized. `subquery()` alias requires `SafeIdentifier`. `existsSql()` wraps in hardcoded `EXISTS (...)`. `whereInSubquery()` combines `SafeIdentifier` field with `Query` subquery.

**Batch UPDATE/DELETE** -- `SetClause.field` requires `SafeIdentifier`. `SetClause.value` uses `SqlPart` (sealed, type-dispatched). `UpdateQuery` and `DeleteQuery` error on empty WHERE clauses, preventing accidental full-table mutations.

**Extended Query Features** -- `NullsPosition` and `CrossJoin` use sealed interfaces with hardcoded keywords. `LockMode` (`ForUpdate`/`ForShare`) uses sealed interfaces with hardcoded strings.

**Changeset Enrichment** -- All validation methods operate on the `changes` table (string validation only, no SQL generation). Field parameters require `SafeIdentifier`. Values only reach SQL through `toInsertSql()`/`toUpdateSql()` which use type-dispatched escaping. Immutable pattern preserved.

**Schema Enrichment** -- `TableDef.primaryKey` uses `SafeIdentifier`. `pkName()` returns `SafeIdentifier.sqlValue` or hardcoded `"id"`. `FieldDef.defaultValue` uses `SqlPart` (sealed). `SqlDefault` renders hardcoded `"DEFAULT"`. Virtual fields are excluded from SQL entirely.

## Getting Started

### Installation

Copy (or vendor) the `orm/`, `std/`, and `temper-core/` directories into your Lua project. Ensure the directories are on your `package.path`:

```lua
-- Add vendor directory to package path
package.path = package.path .. ";vendor/?.lua;vendor/?/init.lua"
```

### Prerequisites

- Lua 5.3+ or LuaJIT
- `lsqlite3` (if using SQLite): `luarocks install lsqlite3`

### Minimal Example

```lua
local orm = require("orm")

-- Define a table
local todos_table = orm.TableDef(
    orm.safeIdentifier("todos"),
    {
        orm.FieldDef(orm.safeIdentifier("title"), orm.StringField(), false, nil, false),
        orm.FieldDef(orm.safeIdentifier("completed"), orm.IntField(), false, nil, false),
    },
    nil
)

-- SELECT query
local select_sql = orm.from(orm.safeIdentifier("todos"))
    :orderBy(orm.safeIdentifier("created_at"), true)
    :limit(50)
    :toSql()
    :toString()
print(select_sql)
-- => SELECT * FROM todos ORDER BY created_at ASC LIMIT 50

-- INSERT via changeset
local params = {title = "Buy milk", completed = "0"}
local cs = orm.changeset(todos_table, params)
    :cast({orm.safeIdentifier("title"), orm.safeIdentifier("completed")})
    :validateRequired({orm.safeIdentifier("title")})

if cs.isValid then
    print(cs:toInsertSql():toString())
    -- => INSERT INTO todos (title, completed) VALUES ('Buy milk', 0)
end
```

## Architecture

This repository is updated automatically by CI. The cascade works as follows:

```
alloy (Temper source)
  |
  | temper build -b lua
  v
temper.out/lua/
  |
  | CI publish (push to this repo)
  v
alloy-lua (this repo)
  |
  | notify-app.yml workflow (triggered on push)
  v
alloy-lua-app (demo application)
  |
  | vendors orm/, std/, temper-core/ into vendor/
  v
Application uses the ORM
```

A push to `main` in the [Alloy source repository](https://github.com/notactuallytreyanastasio/alloy) triggers CI, which builds the Temper compiler, compiles the ORM for all 6 backends, runs the test suite, and pushes the compiled Lua output to this repository. The `notify-app.yml` workflow in this repo then automatically vendors the updated ORM into the [alloy-lua-app](https://github.com/notactuallytreyanastasio/alloy-lua-app) demo application.
